zad4
SELECT
   e.last_name,
   p.product_name,
   SUM(s.quantity * s.price) OVER (PARTITION BY e.employee_id ORDER BY s.sale_date) AS skumulowana_wartosc,
   RANK() OVER (ORDER BY (s.quantity * s.price) DESC) AS ranking_wartosci
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

zad5
SELECT
   e.last_name,
   p.product_name,
   s.price,
   COUNT(s.sale_id) OVER (PARTITION BY p.product_id, TRUNC(s.sale_date)) AS transakcje_dnia,
   SUM(s.quantity * s.price) OVER (PARTITION BY p.product_id, TRUNC(s.sale_date)) AS suma_zapl_dnia,
   LAG(s.price) OVER (PARTITION BY p.product_id ORDER BY s.sale_date) AS poprzednia_cena,
   LEAD(s.price) OVER (PARTITION BY p.product_id ORDER BY s.sale_date) AS kolejna_cena
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

SELECT
   p.product_name,
   s.price,
   SUM(s.quantity * s.price) OVER (PARTITION BY TO_CHAR(s.sale_date, 'YYYY-MM')) AS suma_miesiac_calkowita,
   SUM(s.quantity * s.price) OVER (PARTITION BY p.product_id, TO_CHAR(s.sale_date, 'YYYY-MM') ORDER BY s.sale_date) AS suma_rosnaca
FROM sales s
JOIN products p ON s.product_id = p.product_id;

zad7
WITH DaneRoczne AS (
   SELECT
       p.product_name,
       p.product_category,
       EXTRACT(YEAR FROM s.sale_date) AS rok,
       TO_CHAR(s.sale_date, 'MM-DD') AS dzien_miesiac,
       s.price
   FROM sales s
   JOIN products p ON s.product_id = p.product_id
   WHERE EXTRACT(YEAR FROM s.sale_date) IN (2022, 2023)
),
Porownanie AS (
   SELECT
       product_name,
       product_category,
       rok,
       price AS cena_2023,
       LAG(price) OVER (PARTITION BY product_name, dzien_miesiac ORDER BY rok) AS cena_2022
   FROM DaneRoczne
)
SELECT
   product_category,
   product_name,
   cena_2022,
   cena_2023,
   (cena_2023 - cena_2022) AS roznica_cen
FROM Porownanie
WHERE rok = 2023 AND cena_2022 IS NOT NULL;

zad8
SELECT
   p.product_category,
   p.product_name,
   s.price,
   MIN(s.price) OVER (PARTITION BY p.product_category) AS min_cena_kat,
   MAX(s.price) OVER (PARTITION BY p.product_category) AS max_cena_kat,
   (MAX(s.price) OVER (PARTITION BY p.product_category) - MIN(s.price) OVER (PARTITION BY p.product_category)) AS roznica_cen
FROM sales s
JOIN products p ON s.product_id = p.product_id;

zad9
SELECT
   p.product_name,
   AVG(s.price) 
   OVER 
   (
       PARTITION BY p.product_id
       ORDER BY s.sale_date
       ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
   ) 
   AS srednia_kroczaca
FROM sales s
JOIN products p ON s.product_id = p.product_id;

zad10
SELECT
   p.product_name,
   p.product_category,
   RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking,
   ROW_NUMBER() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS numer_wiersza,
   DENSE_RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking_gesty
FROM sales s
JOIN products p ON s.product_id = p.product_id;
